create table teens (
id serial primary key,
name VARCHAR(50),
gender VARCHAR(3)
);

Используя cross join составить все возможные разнополые пары

SELECT concat (t1.name, ' ',t2.name) as "Пары" FROM teens t1 cross join teens t2
WHERE t1.gender != t2.gender;
